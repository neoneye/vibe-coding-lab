```js
// 2007 speed hacks, preserved. fastSin: parabola approximation.
// fastLog2/fastPow2: float bit reinterpretation (Float32/Int32 views give
// exactly the C++ *(int*)& punning).
function fastSin(x) {
  if (x < -3.14159265) x += 6.28318531;
  else if (x > 3.14159265) x -= 6.28318531;
  const B = 4 / Math.PI, C = -4 / (Math.PI * Math.PI);
  return B * x + C * x * Math.abs(x);
}

const _punF = new Float32Array(1);
const _punI = new Int32Array(_punF.buffer);
const SHIFT23 = 1 << 23;
const OOSHIFT23 = 1.0 / (1 << 23);

function fastLog2(i) {
  _punF[0] = i;
  let x = _punI[0] * OOSHIFT23;
  x -= 127;
  let y = x - Math.floor(x);
  y = (y - y * y) * 0.346607;
  return x + y;
}

function fastPow2(i) {
  let y = i - Math.floor(i);
  y = (y - y * y) * 0.33971;
  let x = i + 127 - y;
  x *= SHIFT23;
  _punI[0] = x;        // Int32Array assignment truncates like the C++ (int) cast
  return _punF[0];
}

function fastPow(a, b) {
  return fastPow2(b * fastLog2(a));
}
```

```js
// ---------------------------------------------------------------------------
// Simon's fixed-point simplex noise (port of simons_simplex_noise_fixedpoint).
// Inputs are 16.16 fixed point. Internals: 12-bit fractions after >>4.
// ---------------------------------------------------------------------------
const I06_MUL_256 = Math.trunc(0.6 * 65536.0 * 256.0);
const I_G3_12 = Math.trunc(4096.0 * 0.166666667);

function _simplexCorner(x, y, z, k, j, i) {
  const t = (I06_MUL_256 - Math.imul(x, x) - Math.imul(y, y) - Math.imul(z, z)) | 0;
  if (t <= 0) return 0;
  const perm0 = PERM512[k];
  const perm1 = PERM512[j + perm0];
  const hash = PERM512[i + perm1];
  const h = hash & 15;
  const u = h < 8 ? x : y;
  const v = h < 4 ? y : (h === 12 || h === 14 ? x : z);
  const ig = ((h & 1) ? -u : u) + ((h & 2) ? -v : v);
  let ut = t >>> 12;
  ut = Math.imul(ut, ut) >>> 12;
  ut = Math.imul(ut, ut) >>> 8;
  return Math.imul(ut, ig) | 0;
}

function simplexNoiseFixed(x, y, z) {
  x |= 0; y |= 0; z |= 0;
  const s = Math.trunc((x + y + z) / 3) | 0;
  const i = ((x + s) | 0) & 0xffff0000;
  const j = ((y + s) | 0) & 0xffff0000;
  const k = ((z + s) | 0) & 0xffff0000;
  const ii = (i >> 16) & 255;
  const jj = (j >> 16) & 255;
  const kk = (k >> 16) & 255;
  const t = Math.trunc((i + j + k) / 6) | 0;
  const x0 = ((x - (i - t)) | 0) >> 4;
  const y0 = ((y - (j - t)) | 0) >> 4;
  const z0 = ((z - (k - t)) | 0) >> 4;

  let i1, j1, k1, i2, j2, k2;
  if (x0 >= y0) {
    if (y0 >= z0) { i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 1; k2 = 0; }
    else if (x0 >= z0) { i1 = 1; j1 = 0; k1 = 0; i2 = 1; j2 = 0; k2 = 1; }
    else { i1 = 0; j1 = 0; k1 = 1; i2 = 1; j2 = 0; k2 = 1; }
  } else {
    if (y0 < z0) { i1 = 0; j1 = 0; k1 = 1; i2 = 0; j2 = 1; k2 = 1; }
    else if (x0 < z0) { i1 = 0; j1 = 1; k1 = 0; i2 = 0; j2 = 1; k2 = 1; }
    else { i1 = 0; j1 = 1; k1 = 0; i2 = 1; j2 = 1; k2 = 0; }
  }

  const x1 = x0 - (i1 << 12) + I_G3_12;
  const y1 = y0 - (j1 << 12) + I_G3_12;
  const z1 = z0 - (k1 << 12) + I_G3_12;
  const x2 = x0 - (i2 << 12) + 2 * I_G3_12;
  const y2 = y0 - (j2 << 12) + 2 * I_G3_12;
  const z2 = z0 - (k2 << 12) + 2 * I_G3_12;
  const x3 = x0 - (1 << 12) + 3 * I_G3_12;
  const y3 = y0 - (1 << 12) + 3 * I_G3_12;
  const z3 = z0 - (1 << 12) + 3 * I_G3_12;

  const n0 = _simplexCorner(x0, y0, z0, kk, jj, ii);
  const n1 = _simplexCorner(x1, y1, z1, kk + k1, jj + j1, ii + i1);
  const n2 = _simplexCorner(x2, y2, z2, kk + k2, jj + j2, ii + i2);
  const n3 = _simplexCorner(x3, y3, z3, kk + 1, jj + 1, ii + 1);
  return ((n0 + n1 + n2 + n3) | 0) >> 7;
}

// ---------------------------------------------------------------------------
// Fixed-point classic Perlin (port of perlin_noise_fixedpoint), INoise.java
// style. Inputs 16.16; lerp mix ratios are 12-bit.
// ---------------------------------------------------------------------------
function perlinFade(t) {
  const t0 = FADE_LUT[t >> 8];
  const t1 = FADE_LUT[Math.min(255, (t >> 8) + 1)];
  return t0 + ((Math.imul(t & 255, t1 - t0)) >> 8);
}

function lerpFixed(t, a, b) {
  return (a + (Math.imul(t, (b - a) | 0) >> 12)) | 0;
}

function _perlinGrad(hash, x, y, z) {
  const h = hash & 15;
  const u = h < 8 ? x : y;
  const v = h < 4 ? y : (h === 12 || h === 14 ? x : z);
  return ((h & 1) === 0 ? u : -u) + ((h & 2) === 0 ? v : -v);
}

function perlinNoiseFixed(x, y, z, fadeZ) {
  x |= 0; y |= 0; z |= 0;
  const X = (x >> 16) & 255;
  const Y = (y >> 16) & 255;
  const Z = (z >> 16) & 255;
  const N = 1 << 16;
  x &= N - 1;
  y &= N - 1;
  z &= N - 1;
  const u = perlinFade(x);
  const v = perlinFade(y);
  const w = fadeZ;
  const A = PERM512[X] + Y;
  const AA = PERM512[A] + Z;
  const AB = PERM512[A + 1] + Z;
  const B = PERM512[X + 1] + Y;
  const BA = PERM512[B] + Z;
  const BB = PERM512[B + 1] + Z;
  return lerpFixed(w,
    lerpFixed(v,
      lerpFixed(u, _perlinGrad(PERM512[AA], x, y, z), _perlinGrad(PERM512[BA], x - N, y, z)),
      lerpFixed(u, _perlinGrad(PERM512[AB], x, y - N, z), _perlinGrad(PERM512[BB], x - N, y - N, z))
    ),
    lerpFixed(v,
      lerpFixed(u, _perlinGrad(PERM512[AA + 1], x, y, z - N), _perlinGrad(PERM512[BA + 1], x - N, y, z - N)),
      lerpFixed(u, _perlinGrad(PERM512[AB + 1], x, y - N, z - N), _perlinGrad(PERM512[BB + 1], x - N, y - N, z - N))
    )
  );
}
```

```js
// ---------------------------------------------------------------------------
// Octave pipeline: port of render_fixedpoint. Per-octave Int32Array cache,
// rebuilt only when seed/frequency/backend/size change; composite() does the
// cheap mode/fadeoff/amplify/gamma/color stage.
// ---------------------------------------------------------------------------
const MAX_OCTAVES = 8;

class PerlinEngine {
  static BACKENDS = {
    simplex: { frequencyFactor: 0.375, amplifyTable: [0.6, 0.6, 0.5, 0.4] },
    perlin:  { frequencyFactor: 0.5,   amplifyTable: [1.0, 0.75, 0.5, 0.4] },
  };

  constructor({ size }) {
    this.size = size;
    this.octaveNoise = [];     // Int32Array per octave, lazily built
    this.validOctaves = 0;
    this.buildCount = 0;
    this.cacheKey = "";
    this.sumBuffer = new Float32Array(size * size);
    this.params = null;
  }

  setParams(p) {
    this.params = p;
    const key = `${p.backend}|${p.seed}|${p.frequency}|${this.size}`;
    if (key !== this.cacheKey) {
      this.cacheKey = key;
      this.validOctaves = 0;
    }
  }

  octavesPending() {
    return Math.max(0, Math.min(this.params.octaves, MAX_OCTAVES) - this.validOctaves);
  }

  // Build the next missing octave; returns true if work was done.
  buildNextOctave() {
    const want = Math.min(this.params.octaves, MAX_OCTAVES);
    if (this.validOctaves >= want) return false;
    this._buildOctave(this.validOctaves);
    this.validOctaves++;
    this.buildCount++;
    return true;
  }

  buildAllOctaves() {
    while (this.buildNextOctave()) { /* build */ }
  }

  _buildOctave(octave) {
    const size = this.size, p = this.params;
    const cfg = PerlinEngine.BACKENDS[p.backend];
    const noiseFn = p.backend === "simplex" ? simplexNoiseFixed : perlinNoiseFixed;
    if (!this.octaveNoise[octave] || this.octaveNoise[octave].length !== size * size) {
      this.octaveNoise[octave] = new Int32Array(size * size);
    }
    const out = this.octaveNoise[octave];

    const frequency = Math.pow(2, p.frequency) * cfg.frequencyFactor;
    const areaSpan = frequency * (1 << octave);
    const ex = 2 * areaSpan;
    const delta = ex / (size - 0.5);
    const seed = p.seed / 400;
    const nz = octave + 1 + seed * 1000;

    const ilower = Math.trunc(-areaSpan * 65536) | 0;
    const idelta = Math.trunc(delta * 65536) | 0;
    const iex = Math.trunc(ex * 65536) | 0;
    const inz = Math.trunc(nz * 65536) | 0;
    const blendInv = Math.trunc((4095 * 65536) / (size - 1)) | 0;
    const fadeZ = p.backend === "perlin" ? perlinFade(inz & 0xffff) : 0;

    let iycur = ilower;
    let d = 0;
    for (let y = 0; y < size; y++) {
      let ixcur = ilower;
      const iblendY = Math.imul(y, blendInv) >> 16;
      for (let x = 0; x < size; x++, d++) {
        const iblendX = Math.imul(x, blendInv) >> 16;
        let bl, br, tl, tr;
        if (p.backend === "simplex") {
          bl = noiseFn(ixcur, iycur, inz);
          br = noiseFn((ixcur + iex) | 0, iycur, inz);
          tl = noiseFn(ixcur, (iycur + iex) | 0, inz);
          tr = noiseFn((ixcur + iex) | 0, (iycur + iex) | 0, inz);
        } else {
          bl = noiseFn(ixcur, iycur, inz, fadeZ);
          br = noiseFn((ixcur + iex) | 0, iycur, inz, fadeZ);
          tl = noiseFn(ixcur, (iycur + iex) | 0, inz, fadeZ);
          tr = noiseFn((ixcur + iex) | 0, (iycur + iex) | 0, inz, fadeZ);
        }
        // original argument order: (blend, br, bl) and (blendY, top, bottom)
        const iy0 = lerpFixed(iblendX, br, bl);
        const iy1 = lerpFixed(iblendX, tr, tl);
        out[d] = lerpFixed(iblendY, iy1, iy0);
        ixcur = (ixcur + idelta) | 0;
      }
      iycur = (iycur + idelta) | 0;
    }
  }

  composite() {
    const size = this.size, p = this.params;
    const cfg = PerlinEngine.BACKENDS[p.backend];
    const octaves = Math.min(this.validOctaves, Math.min(p.octaves, MAX_OCTAVES));
    const fadeoff = p.fadeoff / 100;
    const amplify = (p.amplify / 100) * cfg.amplifyTable[p.mode];
    const gamma = p.gamma / 100;
    const n = size * size;
    const sum = this.sumBuffer;

    for (let i = 0; i < n; i++) {
      let scale = 1, s = 0;
      for (let o = 0; o < octaves; o++) {
        let v = this.octaveNoise[o][i] / 65536;
        switch (p.mode) {
          case 0:
            s += v * scale;
            break;
          case 1:
            if (fadeoff < 0 && (o & 1)) s -= Math.abs(v * scale);
            else s += Math.abs(v * scale);
            break;
          case 2:
            s += fastSin(v * Math.PI * 1.33) * scale;
            break;
          case 3:
            v += fastSin(v * Math.PI * 1.33);
            if (fadeoff < 0 && (o & 1)) s -= Math.abs(v * scale);
            else s += Math.abs(v * scale);
            break;
        }
        scale *= fadeoff;
      }
      sum[i] = s * amplify;
    }

    const out = new Uint8ClampedArray(n * 4);
    const c0 = p.colorBegin, c1 = p.colorEnd;
    for (let i = 0; i < n; i++) {
      let v;
      if (p.mode === 0 || p.mode === 2) v = (sum[i] + 1) * 0.5;   // remap [-1,1]->[0,1]
      else v = sum[i];
      if (v < 0.00001) v = 0.00001;
      v = fastPow(v, gamma);
      out[i * 4] = c0[0] + (c1[0] - c0[0]) * v;
      out[i * 4 + 1] = c0[1] + (c1[1] - c0[1]) * v;
      out[i * 4 + 2] = c0[2] + (c1[2] - c0[2]) * v;
      out[i * 4 + 3] = 255;
    }
    return out;
  }
}
```

```js
"use strict";
const SIZE = 512;
const display = document.getElementById("display-canvas");
const displayCtx = display.getContext("2d");
const offscreen = document.createElement("canvas");
offscreen.width = SIZE;
offscreen.height = SIZE;
const offCtx = offscreen.getContext("2d");
const image = offCtx.createImageData(SIZE, SIZE);

const engine = new PerlinEngine({ size: SIZE });

function resizeCanvas() {
  const wrap = display.parentElement;
  const avail = Math.max(320, Math.min(
    wrap.clientWidth - 26,
    window.innerHeight - display.getBoundingClientRect().top - 40
  ));
  const size = Math.floor(avail);
  if (display.width !== size) {
    display.width = size;
    display.height = size;
    needsPaint = true;
  }
}
window.addEventListener("resize", resizeCanvas);

function hexToRgb(h) {
  return [parseInt(h.slice(1, 3), 16), parseInt(h.slice(3, 5), 16), parseInt(h.slice(5, 7), 16)];
}

const SLIDER_IDS = ["seed", "frequency", "octaves", "fadeoff", "amplify", "gamma"];
function currentParams() {
  const p = {
    backend: document.getElementById("backend").value,
    mode: Number(document.getElementById("mode").value),
    colorBegin: hexToRgb(document.getElementById("color-begin").value),
    colorEnd: hexToRgb(document.getElementById("color-end").value),
  };
  for (const id of SLIDER_IDS) {
    const v = Number(document.getElementById(id).value);
    p[id === "frequency" ? "frequency" : id] = v;
    const el = document.getElementById(id + "-value");
    el.textContent = ["fadeoff", "amplify", "gamma"].includes(id) ? (v / 100).toFixed(2) : v;
  }
  return p;
}

let needsComposite = true;
let needsPaint = true;
function markDirty() {
  engine.setParams(currentParams());
  needsComposite = true;
}
for (const id of SLIDER_IDS) document.getElementById(id).addEventListener("input", markDirty);
for (const id of ["backend", "mode", "color-begin", "color-end"]) {
  document.getElementById(id).addEventListener("input", markDirty);
}
document.getElementById("tile-preview").addEventListener("change", () => { needsPaint = true; });

document.getElementById("reset-btn").addEventListener("click", () => {
  const set = (id, v) => { document.getElementById(id).value = v; };
  set("backend", "simplex"); set("seed", 0); set("frequency", 2); set("octaves", 3);
  set("mode", 0); set("fadeoff", 100); set("amplify", 100); set("gamma", 100);
  set("color-begin", "#000000"); set("color-end", "#ffffff");
  markDirty();
});
document.getElementById("random-btn").addEventListener("click", () => {
  const set = (id, v) => { document.getElementById(id).value = v; };
  set("seed", Math.floor(Math.random() * 401));
  set("frequency", 1 + Math.floor(Math.random() * 6));
  set("octaves", 2 + Math.floor(Math.random() * 7));
  set("mode", Math.floor(Math.random() * 4));
  set("fadeoff", Math.random() < 0.2 ? -Math.floor(40 + Math.random() * 160) : Math.floor(40 + Math.random() * 160));
  set("amplify", 60 + Math.floor(Math.random() * 190));
  set("gamma", 30 + Math.floor(Math.random() * 270));
  markDirty();
});

function paint(pixels) {
  image.data.set(pixels);
  offCtx.putImageData(image, 0, 0);
  displayCtx.imageSmoothingEnabled = true;
  if (document.getElementById("tile-preview").checked) {
    const h = display.width / 2;
    for (const [dx, dy] of [[0, 0], [h, 0], [0, h], [h, h]]) {
      displayCtx.drawImage(offscreen, dx, dy, h, h);
    }
  } else {
    displayCtx.drawImage(offscreen, 0, 0, display.width, display.height);
  }
}

let lastPixels = null;
function frame() {
  if (engine.octavesPending() > 0) {
    engine.buildNextOctave();        // one octave per frame keeps the UI alive
    needsComposite = true;
    document.getElementById("readout-octaves").textContent =
      engine.validOctaves + " / " + Math.min(engine.params.octaves, 8) +
      (engine.octavesPending() > 0 ? " (building)" : "");
  }
  if (needsComposite && engine.octavesPending() === 0) {
    const t0 = performance.now();
    lastPixels = engine.composite();
    document.getElementById("readout-ms").textContent = (performance.now() - t0).toFixed(1) + " ms";
    document.getElementById("readout-octaves").textContent =
      engine.validOctaves + " / " + Math.min(engine.params.octaves, 8);
    needsComposite = false;
    needsPaint = true;
  }
  if (needsPaint && lastPixels) {
    paint(lastPixels);
    needsPaint = false;
  }
  requestAnimationFrame(frame);
}

engine.setParams(currentParams());
resizeCanvas();
requestAnimationFrame(frame);
```
