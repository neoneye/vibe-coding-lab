**Goal:** Build a self-contained `3d-geo-guess/index.html` game where the player is shown a country name or flag and clicks the spot on an interactive globe to guess its location, with 3 health points.

**Architecture:** A single `index.html` renders Earth as a 2D-canvas orthographic globe (no WebGL, no libraries). All pure logic — projection, geometry, flags, game state — lives in a classic `<script id="shared-code">` block so `test.mjs` can extract and run an in-file `GeoTests` harness. Country borders come from Natural Earth 110m, embedded inline by a one-time `build_data.mjs` converter into a `<script id="country-data">` block.

**Tech Stack:** Plain HTML + Canvas 2D + vanilla JS (classic scripts, no modules/imports inside the page). Node (built-in `fetch`) only for the offline converter and the test runner.

## Global Constraints

- The shipped page is a **single self-contained `index.html`** — no CDN links, no external runtime dependencies, no image/font assets. Flags are Unicode emoji only.
- All pure logic goes in **one `<script id="shared-code">` classic script** — **no `import`/`export`/`type="module"`** inside it (so `new Function(...)` in `test.mjs` can run it). Use `function` declarations and `const`/`var` at top level.
- Tests are registered with the in-file **`GeoTests.add(name, fn)`** harness and run via **`node test.mjs`**, which extracts the shared-code block and calls `GeoTests.run()`.
- Country data lives in a **separate `<script id="country-data">window.COUNTRIES=[...]</script>`** block, written by `build_data.mjs`. Game logic receives the country list as a parameter (dependency injection) so tests never need the real data.
- Earth radius constant for distances: **6371 km**. Correct-guess tolerance: **`TOLERANCE_KM = 150`**.

Interfaces defined in the shared-code block (all global, classic script):

- `project(lon, lat, rot) -> {x, y, visible}` — `rot = {lambda, phi}` in degrees; `x,y` in math coords (y up, range ~[-1,1]); `visible` true on the near hemisphere.
- `unproject(x, y, rot) -> {lon, lat} | null` — inverse; `null` when `x²+y² > 1` (outside disk). Always returns a near-hemisphere point.
- `greatCircleKm(a, b) -> number` — haversine km between `[lon,lat]` points.
- `pointInPolygon(lon, lat, ring) -> boolean` — ray-casting test against one ring `[[lon,lat],...]`.
- `distanceToCountryKm(lon, lat, country) -> number` — `0` if inside any ring, else min great-circle distance to any ring vertex.
- `iso2ToEmoji(iso2) -> string` — 2-letter code to flag emoji; `''` for invalid input.
- `createGame(countries, rng?) -> { state, guess, next, restart, pickTarget }` — see Task 4 for the exact state shape and method contracts.
- `TOLERANCE_KM` — number constant `150`.

```html
// ---- projection: orthographic globe (math coords, y up, unit sphere) ----
const DEG = Math.PI / 180;

function project(lon, lat, rot) {
  const a = (lon + rot.lambda) * DEG;
  const b = lat * DEG;
  const cx = Math.cos(b) * Math.sin(a);
  const cy = Math.sin(b);
  const cz = Math.cos(b) * Math.cos(a);
  const p = rot.phi * DEG;
  const ty = cy * Math.cos(p) - cz * Math.sin(p);
  const tz = cy * Math.sin(p) + cz * Math.cos(p);
  return { x: cx, y: ty, visible: tz > 0 };
}

function unproject(x, y, rot) {
  const r2 = x * x + y * y;
  if (r2 > 1) return null;
  const tz = Math.sqrt(1 - r2);
  const ty = y;
  const p = rot.phi * DEG;
  const cy = ty * Math.cos(p) + tz * Math.sin(p);
  const cz = -ty * Math.sin(p) + tz * Math.cos(p);
  const cx = x;
  const lat = Math.asin(Math.max(-1, Math.min(1, cy))) / DEG;
  let lon = Math.atan2(cx, cz) / DEG - rot.lambda;
  lon = ((lon + 180) % 360 + 360) % 360 - 180;
  return { lon, lat };
}
```

```js
// ---- geo: distances and polygon hit-testing ----
function greatCircleKm(a, b) {
  const R = 6371;
  const dLat = (b[1] - a[1]) * DEG, dLon = (b[0] - a[0]) * DEG;
  const la1 = a[1] * DEG, la2 = b[1] * DEG;
  const h = Math.sin(dLat / 2) ** 2 + Math.cos(la1) * Math.cos(la2) * Math.sin(dLon / 2) ** 2;
  return 2 * R * Math.asin(Math.min(1, Math.sqrt(h)));
}

function pointInPolygon(lon, lat, ring) {
  let inside = false;
  for (let i = 0, j = ring.length - 1; i < ring.length; j = i++) {
    const xi = ring[i][0], yi = ring[i][1], xj = ring[j][0], yj = ring[j][1];
    const intersect = ((yi > lat) !== (yj > lat)) &&
      (lon < (xj - xi) * (lat - yi) / (yj - yi) + xi);
    if (intersect) inside = !inside;
  }
  return inside;
}

function distanceToCountryKm(lon, lat, country) {
  for (const ring of country.rings) if (pointInPolygon(lon, lat, ring)) return 0;
  let min = Infinity;
  for (const ring of country.rings)
    for (const v of ring) {
      const d = greatCircleKm([lon, lat], v);
      if (d < min) min = d;
    }
  return min;
}

GeoTests.add('greatCircleKm NYC to London ~5570km', () => {
  const d = greatCircleKm([-74.006, 40.7128], [-0.1276, 51.5074]);
  GeoTests.assertApprox(d, 5570, 60, 'distance');
});

GeoTests.add('pointInPolygon inside and outside a square', () => {
  const sq = [[0, 0], [0, 10], [10, 10], [10, 0], [0, 0]];
  GeoTests.assert(pointInPolygon(5, 5, sq), 'center inside');
  GeoTests.assert(!pointInPolygon(20, 5, sq), 'east outside');
  GeoTests.assert(!pointInPolygon(-1, -1, sq), 'sw outside');
});

GeoTests.add('distanceToCountryKm is 0 inside, >0 outside', () => {
  const country = { rings: [[[0, 0], [0, 10], [10, 10], [10, 0], [0, 0]]] };
  GeoTests.assert(distanceToCountryKm(5, 5, country) === 0, 'inside is 0');
  GeoTests.assert(distanceToCountryKm(40, 5, country) > 1000, 'far away is large');
});
```

```js
// ---- flags: ISO 3166-1 alpha-2 -> regional-indicator emoji ----
function iso2ToEmoji(iso2) {
  if (typeof iso2 !== 'string' || iso2.length !== 2) return '';
  const s = iso2.toUpperCase();
  if (!/^[A-Z]{2}$/.test(s)) return '';
  return String.fromCodePoint(
    0x1F1E6 + (s.charCodeAt(0) - 65),
    0x1F1E6 + (s.charCodeAt(1) - 65)
  );
}
```

## Task 4: Game state machine

**Files:**
- Modify: `3d-geo-guess/index.html` (append to the `<script id="shared-code">` block)

**Interfaces:**
- Consumes: `distanceToCountryKm`, `GeoTests`.
- Produces: `TOLERANCE_KM` (=150) and `createGame(countries, rng?)` returning `{ state, guess, next, restart, pickTarget }`.
  - `state = { health:number, score:number, target:country, prompt:'name'|'flag', phase:'guessing'|'revealed'|'over' }`.
  - `guess(lon,lat)` — only acts when `phase==='guessing'`; returns `{ correct:boolean, distanceKm:number }` (or `null` if not guessing). Correct when `distanceKm <= TOLERANCE_KM`: increments `score`; otherwise decrements `health`. Sets `phase` to `'over'` if `health<=0`, else `'revealed'`.
  - `next()` — only from `'revealed'`, picks a new target and sets `phase='guessing'`.
  - `restart()` — resets `health=3, score=0`, picks a new target.
  - `pickTarget()` — picks a random country and a random prompt mode (`'flag'` only when the country has a non-empty `iso2`, otherwise `'name'`); sets `phase='guessing'`.
  - `rng` defaults to `Math.random`; injectable for deterministic tests.

- [ ] **Step 1: Write the function and failing tests**

Append inside the shared-code block (before `</script>`):

```js
// ---- game: health/score state machine (rendering-independent) ----
const TOLERANCE_KM = 150;

function createGame(countries, rng) {
  rng = rng || Math.random;
  const state = { health: 3, score: 0, target: null, prompt: 'name', phase: 'guessing' };
  function pickTarget() {
    const c = countries[Math.floor(rng() * countries.length)];
    state.target = c;
    state.prompt = (rng() < 0.5 && c.iso2) ? 'flag' : 'name';
    state.phase = 'guessing';
  }
  function guess(lon, lat) {
    if (state.phase !== 'guessing') return null;
    const d = distanceToCountryKm(lon, lat, state.target);
    const correct = d <= TOLERANCE_KM;
    if (correct) state.score++; else state.health--;
    state.phase = state.health <= 0 ? 'over' : 'revealed';
    return { correct: correct, distanceKm: d };
  }
  function next() { if (state.phase === 'revealed') pickTarget(); }
  function restart() { state.health = 3; state.score = 0; pickTarget(); }
  pickTarget();
  return { state, guess, next, restart, pickTarget };
}

// Deterministic rng helper for tests: cycles through the given values.
function _seqRng(values) { let i = 0; return () => values[i++ % values.length]; }
const _squareland = [{ name: 'Squareland', iso2: '', rings: [[[0,0],[0,10],[10,10],[10,0],[0,0]]], centroid: [5,5] }];

GeoTests.add('correct guess keeps health, raises score, reveals', () => {
  const g = createGame(_squareland, _seqRng([0, 0.9]));
  const r = g.guess(5, 5);
  GeoTests.assert(r.correct, 'should be correct');
  GeoTests.assert(g.state.health === 3, 'health kept');
  GeoTests.assert(g.state.score === 1, 'score up');
  GeoTests.assert(g.state.phase === 'revealed', 'revealed');
});

GeoTests.add('wrong guess drops health and reveals', () => {
  const g = createGame(_squareland, _seqRng([0, 0.9]));
  const r = g.guess(80, 80);
  GeoTests.assert(!r.correct, 'should be wrong');
  GeoTests.assert(g.state.health === 2, 'health -1');
  GeoTests.assert(g.state.phase === 'revealed', 'revealed');
});

GeoTests.add('three wrong guesses ends the game', () => {
  const g = createGame(_squareland, _seqRng([0, 0.9]));
  g.guess(80, 80); g.next();
  g.guess(80, 80); g.next();
  const r = g.guess(80, 80);
  GeoTests.assert(g.state.health === 0, 'health 0');
  GeoTests.assert(g.state.phase === 'over', 'game over');
  GeoTests.assert(g.guess(5, 5) === null, 'no guessing after over');
});

GeoTests.add('restart resets health and score', () => {
  const g = createGame(_squareland, _seqRng([0, 0.9]));
  g.guess(80, 80);
  g.restart();
  GeoTests.assert(g.state.health === 3 && g.state.score === 0, 'reset');
  GeoTests.assert(g.state.phase === 'guessing', 'guessing again');
});
```

```js
// One-time converter: download Natural Earth 110m admin-0 countries (public domain),
// simplify, and embed as window.COUNTRIES in index.html's country-data block.
// Usage: node build_data.mjs
import { readFileSync, writeFileSync } from "node:fs";
import { fileURLToPath } from "node:url";
import { dirname, join } from "node:path";

const URL = "https://raw.githubusercontent.com/nvkelso/natural-earth-vector/master/geojson/ne_110m_admin_0_countries.geojson";
const round2 = (n) => Math.round(n * 100) / 100;

const res = await fetch(URL);
if (!res.ok) { console.error("download failed:", res.status); process.exit(1); }
const gj = await res.json();

const out = [];
for (const f of gj.features) {
  const p = f.properties;
  const name = p.NAME || p.NAME_LONG || p.ADMIN;
  if (!name || name === "Antarctica") continue;
  let iso2 = (p.ISO_A2_EH && p.ISO_A2_EH !== "-99") ? p.ISO_A2_EH
           : (p.ISO_A2 && p.ISO_A2 !== "-99") ? p.ISO_A2 : "";
  const g = f.geometry;
  if (!g) continue;
  const polys = g.type === "Polygon" ? [g.coordinates]
              : g.type === "MultiPolygon" ? g.coordinates : [];
  const rings = [];
  for (const poly of polys) {
    const outer = poly[0];
    if (!outer || outer.length < 4) continue;
    rings.push(outer.map(([x, y]) => [round2(x), round2(y)]));
  }
  if (!rings.length) continue;
  let big = rings[0];
  for (const r of rings) if (r.length > big.length) big = r;
  let sx = 0, sy = 0;
  for (const v of big) { sx += v[0]; sy += v[1]; }
  const centroid = [round2(sx / big.length), round2(sy / big.length)];
  out.push({ name, iso2, rings, centroid });
}

const json = JSON.stringify(out);
const file = join(dirname(fileURLToPath(import.meta.url)), "index.html");
let html = readFileSync(file, "utf8");
const block = `<script id="country-data">window.COUNTRIES = ${json};</script>`;
html = html.replace(/<script id="country-data">[\s\S]*?<\/script>/, block);
writeFileSync(file, html);
console.log("embedded", out.length, "countries;", json.length, "bytes");
```

- [ ] **Step 2: Add the UI/render script**

Immediately before `</body>` (after the shared-code `<script>` block), add this script:

```html
<script>
(() => {
  const canvas = document.getElementById('globe');
  const ctx = canvas.getContext('2d');
  const W = canvas.width, H = canvas.height;
  const R = Math.min(W, H) / 2 - 6, CX = W / 2, CY = H / 2;
  const rot = { lambda: 0, phi: -15 };
  let anim = null, clickMark = null;

  const game = createGame(window.COUNTRIES);
  const els = {
    prompt: document.getElementById('prompt'),
    hearts: document.getElementById('hearts'),
    score: document.getElementById('score'),
    result: document.getElementById('result'),
    next: document.getElementById('nextBtn'),
    overlay: document.getElementById('overlay'),
    finalScore: document.getElementById('finalScore'),
    restart: document.getElementById('restartBtn'),
  };

  function toScreen(lon, lat) {
    const p = project(lon, lat, rot);
    return { x: CX + p.x * R, y: CY - p.y * R, visible: p.visible };
  }

  function drawCountry(c, fill, stroke) {
    for (const ring of c.rings) {
      ctx.beginPath();
      let started = false;
      for (const [lon, lat] of ring) {
        const s = toScreen(lon, lat);
        if (!s.visible) { started = false; continue; }
        if (!started) { ctx.moveTo(s.x, s.y); started = true; }
        else ctx.lineTo(s.x, s.y);
      }
      ctx.closePath();
      if (fill) { ctx.fillStyle = fill; ctx.fill(); }
      if (stroke) { ctx.strokeStyle = stroke; ctx.lineWidth = 0.6; ctx.stroke(); }
    }
  }

  function render() {
    ctx.clearRect(0, 0, W, H);
    ctx.beginPath(); ctx.arc(CX, CY, R, 0, Math.PI * 2);
    ctx.fillStyle = '#16335c'; ctx.fill();
    for (const c of window.COUNTRIES) drawCountry(c, '#2f6b3a', '#16335c');
    if (game.state.phase !== 'guessing' && game.state.target)
      drawCountry(game.state.target, '#f5c542', '#7a5b00');
    if (clickMark) {
      const s = toScreen(clickMark.lon, clickMark.lat);
      if (s.visible) {
        ctx.beginPath(); ctx.arc(s.x, s.y, 6, 0, Math.PI * 2);
        ctx.fillStyle = clickMark.correct ? '#22c55e' : '#ef4444';
        ctx.strokeStyle = 'white'; ctx.lineWidth = 2; ctx.fill(); ctx.stroke();
      }
    }
  }

  function syncHud() {
    const h = game.state.health;
    els.hearts.textContent = '❤️'.repeat(h) + '\u{1F90D}'.repeat(Math.max(0, 3 - h));
    els.score.textContent = 'Score: ' + game.state.score;
    const t = game.state.target;
    els.prompt.textContent = (game.state.prompt === 'flag') ? iso2ToEmoji(t.iso2) : t.name;
    els.prompt.style.fontSize = (game.state.prompt === 'flag') ? '64px' : '34px';
  }

  function flyTo(centroid, then) {
    const start = { lambda: rot.lambda, phi: rot.phi };
    const target = { lambda: -centroid[0], phi: centroid[1] };
    let dl = ((target.lambda - start.lambda + 540) % 360) - 180;
    const steps = 28; let i = 0;
    if (anim) cancelAnimationFrame(anim);
    (function step() {
      i++;
      const t = i / steps, e = t < 1 ? 1 - Math.pow(1 - t, 3) : 1;
      rot.lambda = start.lambda + dl * e;
      rot.phi = start.phi + (target.phi - start.phi) * e;
      render();
      if (i < steps) anim = requestAnimationFrame(step); else if (then) then();
    })();
  }

  function newRound() {
    clickMark = null;
    els.result.textContent = 'Drag to spin · click the country’s location';
    els.next.style.visibility = 'hidden';
    syncHud(); render();
  }

  function onGuess(lon, lat) {
    const r = game.guess(lon, lat);
    if (!r) return;
    clickMark = { lon, lat, correct: r.correct };
    syncHud();
    flyTo(game.state.target.centroid, render);
    const name = game.state.target.name;
    els.result.textContent = r.correct
      ? '✅ Correct! ' + name
      : '❌ ' + name + ' — ' + Math.round(r.distanceKm) + ' km off';
    if (game.state.phase === 'over') {
      els.finalScore.textContent = 'Final score: ' + game.state.score;
      els.overlay.classList.add('show');
    } else {
      els.next.style.visibility = 'visible';
    }
  }

  // drag to rotate; a click without dragging is a guess
  let down = null, moved = false;
  canvas.addEventListener('pointerdown', (e) => {
    down = { x: e.offsetX, y: e.offsetY, lambda: rot.lambda, phi: rot.phi };
    moved = false; canvas.setPointerCapture(e.pointerId);
  });
  canvas.addEventListener('pointermove', (e) => {
    if (!down) return;
    const dx = e.offsetX - down.x, dy = e.offsetY - down.y;
    if (Math.abs(dx) + Math.abs(dy) > 4) moved = true;
    rot.lambda = down.lambda + dx * 0.4;
    rot.phi = Math.max(-89, Math.min(89, down.phi + dy * 0.4));
    render();
  });
  canvas.addEventListener('pointerup', (e) => {
    const wasDrag = moved; down = null;
    if (wasDrag || game.state.phase !== 'guessing') return;
    const u = unproject((e.offsetX - CX) / R, -(e.offsetY - CY) / R, rot);
    if (u) onGuess(u.lon, u.lat);
  });

  els.next.addEventListener('click', () => { game.next(); newRound(); });
  els.restart.addEventListener('click', () => {
    els.overlay.classList.remove('show'); game.restart(); newRound();
  });

  newRound();
})();
</script>
```
